var a = 1, b = 1, c, d;
c = ++a; alert(c);           // 2 c=1+1, a=1+1 в c кладется результат вычисления
d = b++; alert(d);           // 1 d=1, b=1+1 в d кладется b, потом идёт вычисление нового значения b
c = (2+ ++a); alert(c);      // 5 с=2+(2+1), a=2+1, в c кладем уже инкрементированное значение a
d = (2+ b++); alert(d);      // 4 d=2+(2), b=2+1 сначала сложение, потом вычисление нового значения
alert(a);                    // 3 два раза увеличили переменную на 1, поэтому 3
alert(b);                    // 3 аналогично
